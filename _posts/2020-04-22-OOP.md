```python
class Solution(object):
    def getLeastNumbers(self, arr, k):
        """
        :type arr: List[int]
        :type k: int
        :rtype: List[int]
        """
        new_arr = []
        if k != 0 :
            if k <= len(arr):
                for i in range(0,k):
                    x = min(arr)
                    new_arr.append(x)
                    arr.remove(x)
   
                return new_arr
        else:
                return  new_arr
```
